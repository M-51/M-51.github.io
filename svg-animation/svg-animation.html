SVG Animation simple examples

1. Translate


<script src="svganimation.min.js"></script>

<svg viewBox="-5 -50 200 100" xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" version="1.1" class="svg">
    <rect id="ex1" x="-3" y="-3" width="6" height="6" fill="#444" />
</svg>

<script>
const ex1 = new SVGAnimation();
ex1.add({
    object: '#ex1',
    animation: {
        transform: {
            translate: {
                x: t => t,
                y: t => 10 * Math.sin(t),
            }
        }
    }
})    
</script>

3. Translate, rotate and scale


<script src="svganimation.min.js"></script>

<svg viewBox="-5 -50 100 100" xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" version="1.1">
    <rect id="ex3" x="-3" y="-3" width="6" height="6" fill="#444" />
</svg>

<script>
const ex3 = new SVGAnimation();
ex3.add({
    object: '#ex3',
    animation: {
        transform: {
            translate: {
                x: t => 10 * t,
                y: t => 10 * Math.sin(t),
            },
            rotate: t => t,
            scale: t => Math.sin(t) + 1,
        }
    }
})
</script>

4. Other attributes (stroke-width)


<script src="svganimation.min.js"></script>

<svg viewBox="-5 -50 100 100" xmlns="http://www.w3.org/2000/svg" xmlns:xlink="http://www.w3.org/1999/xlink" version="1.1">
    <rect id="ex4" x="0" y="0" width="20" height="20" fill="none" stroke="#444" stroke-width="1" />
</svg>

<script>
const ex4 = new SVGAnimation();
ex4.add({
    object: '#ex4',
    animation: {
        'stroke-width': {
            value: t => Math.sin(t) + 1,
        }
    }
})
</script>
